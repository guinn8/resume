\documentclass[11pt]{letter}

\usepackage{geometry}
\geometry{left=1in, right=1in, top=1in, bottom=1in}

\usepackage{hyperref}
\usepackage{fontspec}
\usepackage{setspace}

\setmainfont{TeX Gyre Heros} % You can change this to your preferred font

% Redefine styles to align address and closing to the left
\makeatletter
% Align address block to the left
\def\addressstyle{\leftskip0pt\parindent0pt\raggedright}

% Align closing to the left
\renewcommand{\closing}[1]{%
  \par\nobreak\vspace{\parskip}%
  \stopbreaks%
  \noindent%
  #1,\\[2\parskip]%
  \ifx\@empty\fromsig\else \fromsig\fi\par}
\makeatother

\address{
    \textbf{$name$} \\
    $address$ \\
    \href{mailto:$email$}{$email$} \\
    $phone$
}

\signature{$name$}

\begin{document}
\begin{letter}{}

\date{$date$}
\opening{Dear $recipient_name$,}

$body$

\closing{Sincerely}

\end{letter}
\end{document}
